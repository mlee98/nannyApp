use nannyApplication;

create table login_info (
 username varchar(50), 
 password varchar(50), 
 account_type varchar(50), 
 user_id int
);


create table parent_info (
 user_id int,
 name varchar(100), 
 address varchar(200), 
 phone_number varchar(50),
 email varchar(100)
);


create table children_info (
 user_id int,
 age int,
 name varchar(100), 
 gender varchar(10),
 hobbies varchar(100), 
 allergies varchar(100),
 medications varchar(100)
);


create table babysitter_info(
 user_id int,
 name varchar(100), 
 address varchar(200), 
 phone_number varchar(50),
 babysitter_bio varchar (255),
 email varchar(100),
 years_of_experience int,
 age int,
 gender varchar(100)
); 

create table sitter_skills(
 user_id int,
 cpr boolean,
 pet_friendly boolean,
 can_drive boolean,
 can_cook boolean
 );
