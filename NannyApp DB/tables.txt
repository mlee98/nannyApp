use nannyApplication;

create table login_info
(
	username varchar(50),
    password varchar(50),
    account_type varchar(50),
    user_id int
);

create table parent_info
(
	user_id int NOT NULL AUTO_INCREMENT,
    account_type varchar(50),
    username varchar(50),
    password varchar(50),
    first_name varchar(50),
    last_name varchar(50),
    age int,
    gender varchar(50),
    address varchar(50),
    city varchar(50),
    state varchar(50),
    zip int,
    email varchar(100),
    phone_number int,
    primary key (user_id)
);

create table child_info
(
	user_id int,
    name varchar(50),
    gender varchar(50),
    age int,
    likes varchar(250),
    allergies varchar(250),
    medications varchar(250),
    special_requirements varchar(250)
);
    
create table nanny_info
(
	user_id int NOT NULL AUTO_INCREMENT,
    account_type varchar(50),
    username varchar(50),
    password varchar(50),
    first_name varchar(50),
    last_name varchar(50),
    age int,
    gender varchar(50),
    address varchar(50),
    city varchar(50),
    state varchar(50),
    zip int,
    email varchar(100),
    phone_number int,
    primary key (user_id)
);    
    
create table nanny_references
(
		user_id int,
        name varchar(50),
        phone_number int,
        email varchar(50)
);
  
create table skills
(
	user_id int,
    cpr bool,
    can_cook bool,
    pet_friendly bool,
    can_drive bool
);
